def performLinearRegression(iteratn, features, sampSize):
    significant_percent = []
    total_Count = iteratn*(features + 1) #+1 for intercept
    
    for r in range(1, iteratn+1): #Repeat R times
        ## Create random data and estimate a linear regression model w/statsmodels
        x = np.random.normal(size=(sampSize, features))
        eps = np.random.normal(size=(sampSize, 1))
        y = eps
        X = np.concatenate((np.ones_like(y), x), axis=1)
        model = smf.OLS(y, X)
        fitted_model = model.fit()
        
        #Statistical significance and p-values
        t_values = fitted_model.tvalues
        p_values = stats.t.sf(np.abs(fitted_model.tvalues), sampSize-features-1)*2
        significant_percent.append(np.mean(np.abs(t_values) > 1.96))
       
    return significant_percent

np.random.seed(25)
no_of_rep = 1000
spanish_speaking_list = []

def find_spanish_speaking():
    for i in range(num_of_Rep):
        rand_sample = bernoulli.rvs(size=100,p=0.3)
        num_speaking = (rand_sample == 1).sum()
        spanish_speaking_list.append(num_speaking)
    no_of_times_lt_25 = sum(i < 25 for i in spanish_speaking_list)  
    return no_of_times_lt_25
    
spanish_speaking_folks = find_spanish_speaking()
print("No of random samples with less than 25 spanish speaking using bernoulli: " + str(spanish_speaking_folks))

By comparing the simulation results from Bernoulli and Binomial distribution, we could see that the number of random samples that has less than 25 spanish speaking students is slightly higher in Binomial distribution(119) than in Bernoulli distribution(105). In bernoulli distribution, the probability is expressed as p and 1-p. But in case of Binomial distribution, there are n independent trials and the probability of getting 1 t times is expressed as (p^t)*((1-p)^(n-t)). This might give us a higher probability. A binomial random variable with parameter n=1 is equivalent to a Bernoulli random variable, i.e. there is only one trial.¶

num_of_Rep = 1000
actual_max = 71
highest_list = []

np.random.seed(90)
def calculate_probability(n):
    np.random.seed(90)
    highest_list = []
    for i in range(1, num_of_Rep+1):
        rsample = random.sample(range(1,n), 3)
        highest_list.append(max(rsample))

    no_of_max_lt_71 = sum(i <= actual_max for i in highest_list)
    prob_max_lt_71 = no_of_max_lt_71/1000
    return prob_max_lt_71


prob_n_100 = calculate_probability(100)
prob_n_200 = calculate_probability(200)
