def buildModel(x, n):

    mNN = KNeighborsClassifier(n_neighbors = n).fit(x, y)

    ## Predict on a 10x10 grid
    range1 = np.linspace(4, 8, 100)
    range2 = np.linspace(2, 5, 100)
    (x02d, x12d) = np.meshgrid(range1, range2)
    # create a grid (two matrices)
    x0 = x02d.flatten()
    # create vector of grid variables
    x1 = x12d.flatten()
    xp = np.column_stack((x0, x1))
    # stack these variables next to each otehr
    ypNN = mNN.predict(xp)
    #yPNN_1 = KNN_1.predict(xp)
    # predict
    ypNN2d = ypNN.reshape((100,100))
    #yPNN_12d = yPNN_1.reshape((100,100))
    # mold it back to a matrix
    
    ## plot
    # use plt.imshow() to show image.  Use alpha for transparency
    plt.imshow(ypNN2d, extent=[4, 8, 2, 5], alpha = 0.5,
               origin='lower', aspect='auto')
    # use plt.contour to draw decision boundary
    plt.contour(x02d, x12d, ypNN2d, levels=[0.5], colors='black')
    plt.scatter(x[:,0], x[:,1], c=y, alpha=0.5, s=100)
